Main Python Agent PYTHON 
#!/usr/bin/env python3
"""
KMC-UCDL Daily Facebook Content Agent
Developed by: Mr. Bishnu Singh Rayamajhi
Math Lecturer & Head of Humanity & Social Science, KMC-Taulihawa
Founder: UCDL Academy
"""

import json
import requests
import os
from datetime import datetime
from dotenv import load_dotenv
import logging

load_dotenv()

# Configuration
DEVELOPER = "Mr. Bishnu Singh Rayamajhi"
ORGANIZATION = "KMC-Taulihawa"
FOUNDED_ACADEMY = "UCDL Academy"

logging.basicConfig(level=logging.INFO)

class NewsAggregator:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://newsapi.org/v2/everything"
    
    def fetch_kmc_news(self, limit=5):
        params = {
            'q': 'Kathmandu Metropolitan City OR KMC',
            'sortBy': 'publishedAt',
            'language': 'en',
            'apiKey': self.api_key,
            'pageSize': limit
        }
        response = requests.get(self.base_url, params=params, timeout=10)
        return response.json().get('articles', [])

class ContentGenerator:
    def __init__(self, api_key, model="gpt-4"):
        self.api_key = api_key
        self.model = model
    
    def generate_post(self, articles):
        articles_text = "\n\n".join([
            f"📰 {a['title']}\n{a['description']}" 
            for a in articles[:3]
        ])
        
        payload = {
            "model": self.model,
            "messages": [
                {
                    "role": "system",
                    "content": "तपाईं KMC को social media content creator हुनुहुन्छ।"
                },
                {
                    "role": "user",
                    "content": f"यी articles बाट Facebook post बनाउनुहोस्:\n{articles_text}"
                }
            ],
            "temperature": 0.7
        }
        
        headers = {"Authorization": f"Bearer {self.api_key}"}
        response = requests.post(
            "https://api.openai.com/v1/chat/completions",
            json=payload, headers=headers, timeout=30
        )
        return response.json()['choices'][0]['message']['content']

class FacebookPoster:
    def __init__(self, access_token, page_id):
        self.access_token = access_token
        self.page_id = page_id
    
    def post(self, message, link=None, picture=None):
        url = f"https://graph.facebook.com/v18.0/{self.page_id}/feed"
        data = {
            'message': message,
            'access_token': self.access_token
        }
        if link: data['link'] = link
        if picture: data['picture'] = picture
        
        response = requests.post(url, data=data, timeout=15)
        return response.json()

class KMCFacebookAgent:
    def __init__(self):
        self.news = NewsAggregator(os.getenv('NEWS_API_KEY'))
        self.content = ContentGenerator(
            os.getenv('OPENAI_API_KEY'),
            os.getenv('OPENAI_MODEL', 'gpt-4')
        )
        self.facebook = FacebookPoster(
            os.getenv('FACEBOOK_ACCESS_TOKEN'),
            os.getenv('FACEBOOK_PAGE_ID')
        )
    
    def run_daily_workflow(self):
        # 1. Fetch news
        articles = self.news.fetch_kmc_news(limit=5)
        if not articles:
            return {'status': 'failed', 'error': 'No articles'}
        
        # 2. Generate content
        post_content = self.content.generate_post(articles)
        
        # 3. Post to Facebook
        result = self.facebook.post(
            message=post_content,
            link=articles[0]['url'],
            picture=articles[0].get('urlToImage')
        )
        
        return {
            'status': 'success',
            'post_id': result.get('id'),
            'articles': len(articles),
            'developer': DEVELOPER,
            'organization': ORGANIZATION
        }

# Main execution
if __name__ == "__main__":
    print(f"Developed by: {DEVELOPER}")
    print(f"Organization: {ORGANIZATION}")
    print(f"Founder: {FOUNDED_ACADEMY}\n")
    
    agent = KMCFacebookAgent()
    result = agent.run_daily_workflow()
    print(json.dumps(result, indent=2))

🔧 Troubleshooting

❌
Problem: "Token Expired" (Facebook)
Solution: नयाँ token लिनुहोस् sntr.io/fb-token बाट र .env मा update गर्नुहोस्, फेरि docker-compose restart n8n गर्नुहोस्।
⚠️
Problem: "Rate Limit Exceeded" (OpenAI)
Solution: billing overview check गर्नुहोस्। Batch size घटाउनुहोस् वा plan upgrade गर्नुहोस्।
ℹ️
Problem: "Database Connection Error"
Solution: docker-compose logs postgres run गर्नुहोस् र PostgreSQL logs check गर्नुहोस्।
